```graphql
input DemandHqCreateInput {
  defaultCurrency: ResonanceCurrency!
  defaultLocaleType: ResonanceLocaleType!
  demandSubscriptionCodes: [DemandSubscriptionCode!]
  description: String
  name: String!
  notificationUserNames: DemandHqNotificationUserNamesInput
}
```
