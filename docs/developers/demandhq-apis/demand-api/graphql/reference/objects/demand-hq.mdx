The parent-most entity of the DemandHq Domain


```graphql
type DemandHq {
  createdAt: String!
  defaultCurrency: ResonanceCurrency!
  defaultLocaleType: ResonanceLocaleType!
  demandHqId: ID!
  demandHqIdentities(
  after: String
  before: String
  first: Int
  identityParentEntityId: ID
  identityType: DemandHqIdentityType
  last: Int
): DemandHqIdentityConnection!
  demandHqIdentity(
  demandHqId: ID!
  identityId: ID!
): DemandHqIdentity
  demandSubscriptionCodes: [DemandSubscriptionCode!]
  description: String
  name: String!
  notificationUserNames: DemandHqNotificationUserNames
  ownerUserName: String!
  page(
  pageId: ID
  slug: String
): Page!
  pages(
  after: String
  before: String
  first: Int
  last: Int
  pageType: String
  status: PageStatus
): PageConnection!
  storefront(
  storefrontId: ID!
): Storefront
  storefronts(
  after: String
  before: String
  first: Int
  last: Int
): StorefrontConnection
  updatedAt: String!
  updatedBy: ID!
}
```
