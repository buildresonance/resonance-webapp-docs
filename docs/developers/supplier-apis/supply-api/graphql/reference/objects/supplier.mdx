The parent-most entity of the Supplier Domain


```graphql
type Supplier {
  createdAt: String!

  ownerUserName: String!
  quickLink(
  appId: ID!
  quickLinkType: SupplierQuickLinkType!
): SupplierQuickLink
  quickLinks(
  after: String
  before: String
  first: Int
  last: Int
  quickLinkType: SupplierQuickLinkType!
): SupplierQuickLinkConnection!
  status: SupplierStatus!
  supplierId: ID!
  supplierIdentities(
  after: String
  before: String
  first: Int
  identityParentEntityId: ID
  identityType: SupplierIdentityType
  last: Int
): SupplierIdentityConnection!
  supplierIdentity(
  identityId: ID!
  supplierId: ID!
): SupplierIdentity
  supplierProduct(
  supplierProductId: ID!
): SupplierProduct
  supplierProducts(
  after: String
  before: String
  first: Int
  last: Int
): SupplierProductConnection!
  updatedAt: String!
  updatedBy: ID!
}
```
